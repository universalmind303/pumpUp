export default connect(mapState, mapDispatch)(Header)
 
const {height, width} = Dimensions.get('window')
 
function Header ( {
  user: {
    isNotStarted,

}
 
function mapState(state) {
  return {

}
 
function mapDispatch(dispatch) {
